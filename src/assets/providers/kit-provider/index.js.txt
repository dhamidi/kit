import { Glob } from 'bun'
import { stat } from 'node:fs/promises'

/**
 * KitProviderProvider manages Kit provider modules under providers/<name>/index.js.
 */
class KitProviderProvider {
	constructor(kit) {
		this.kit = kit
	}

	name() {
		return 'kit-provider'
	}

	async *types() {
		yield new KitProviderType(this.kit)
	}

	async *components() {
		const root = await discoveryRoot(this.kit)

		for (const directory of this.kit.providerDiscoveryPaths(root, process.cwd())) {
			if (!(await exists(directory))) {
				continue
			}

			for await (const path of new Glob('*/index.js').scan({ cwd: directory.path() })) {
				yield new KitProviderComponent({ path: directory.join(path).path(), kit: this.kit })
			}
		}
	}

	create(spec, env) {
		return new KitProviderType(this.kit).create(spec, env)
	}

	async formatComponentShow(componentName, record, output) {
		const table = new this.kit.TableFormatter(['Field', 'Value'], output)
		const spec = record.component.inspect()

		table.row(['Component', componentName])
		table.row(['Description', record.component.description()])
		table.row(['Provider name', spec.name])
		table.row(['Project', spec.project])
		table.row(['Files', spec.files.join('\n')])
		table.row(['Spec', `kit component spec ${componentName}`])
		table.row(['Bootstrap', `kit component spec ${componentName} | kit generate kit-provider provider --spec -`])
		table.flush()
	}
}

/**
 * KitProviderType describes how to generate a new Kit provider skeleton.
 */
class KitProviderType {
	constructor(kit) {
		this.kit = kit
	}

	id() {
		return 'provider'
	}

	description() {
		return 'A Kit provider module in providers/<name>/index.js'
	}

	schema() {
		const { Type } = this.kit

		return Type.Object({
			name: Type.String({
				description:
					'Kebab-case provider name without the kit- prefix unless it is part of the name',
				examples: ['sveltekit', 'database', 'kit-widget'],
				pattern: '^[a-z][a-z0-9]*(?:-[a-z0-9]+)*$',
			}),
			description: Type.String({
				description: 'Short user-facing description of the provider',
				examples: ['Generate SvelteKit routes and remote functions'],
			}),
			project: Type.Optional(
				Type.String({
					description:
						'Directory to generate the provider into; defaults to <repo-root>/providers',
					examples: ['providers', '.kit/providers'],
				}),
			),
		})
	}

	parse(argv) {
		return this.kit.parseArgs({
			args: argv,
			options: this.kit.parseArgsOptionsFromSchema(this.schema()),
			strict: true,
			allowPositionals: true,
		})
	}

	describe(spec) {
		return spec.description
	}

	async *create(spec, env) {
		const base = spec.project ?? (await this.kit.repoRoot()).join('providers').path()
		const root = this.kit.FileURI.fromPath(base)
		const path = root.join(spec.name, 'index.js').path()

		yield await env.createFile(path, providerTemplate(spec))
		yield this.kit.Event.plan(
			`Finish generated ${spec.name} provider`,
			[
				{
					id: 'analyze-provider-domain',
					instructions: 'Analyze the target domain and summarize implementation requirements',
					files: [path],
					agent: {
						prompt: analysisPrompt(spec, path),
					},
				},
				{
					id: 'fill-in-provider',
					instructions: 'Use the configured agent to finish generated provider implementation',
					files: [path],
					agent: {
						prompt: providerPrompt(spec, path),
					},
				},
				{
					id: 'verify-provider',
					instructions: 'Verify generated provider is discovered',
					verifyWithCommand: 'bun run kit provider list',
				},
			],
			{ intent: spec.intent },
		)
	}
}

/**
 * KitProviderComponent represents one provider module already present in the checkout.
 */
class KitProviderComponent {
	constructor({ path, kit }) {
		this.path = path
		this.kit = kit
	}

	provider() {
		return 'kit-provider'
	}

	id() {
		return providerName(this.kit.FileURI.fromPath(this.path))
	}

	description() {
		return `Kit provider ${this.id()}`
	}

	inspect() {
		const file = this.kit.FileURI.fromPath(this.path)

		return {
			name: this.id(),
			description: this.description(),
			project: providerProject(file, this.kit.FileURI.fromPath(process.cwd())),
			files: [this.path],
		}
	}
}

function providerName(file) {
	return file.parent().relativeTo(file.parent().parent())
}

function providerProject(file, cwd) {
	try {
		return file.parent().parent().relativeTo(cwd)
	} catch {
		return file.parent().parent().path()
	}
}

/**
 * ${className(spec.name)}Component represents one existing ${spec.name} component.
 */
class ${className(spec.name)}Component {
	constructor({ name, description, files = [] }) {
		// AGENT: Store source-derived facts only. Do not compute generation-specific
		// defaults here; Kit handles schema defaults during normalization.
		this.name = name
		this.componentDescription = description ?? name
		this.files = files
	}

	provider() {
		return ${JSON.stringify(spec.name)}
	}

	id() {
		return this.name
	}

	description() {
		return this.componentDescription
	}

	inspect() {
		// AGENT: This object must conform to ${className(spec.name)}Type.schema().
		// Include files for observability; Kit treats files as a common schema field.
		// Verify with: kit provider test ${spec.name}
		return {
			name: this.name,
			description: this.componentDescription,
			files: this.files,
		}
	}
}

export default function provider(kit) {
	return new ${className(spec.name)}Provider(kit)
}
`
}

function analysisPrompt(spec, path) {
	return `Intent: ${spec.intent ?? spec.description}

Analyze what the generated ${spec.name} provider in ${path} must implement.
The implementation agent may not have Kit source access, so produce a concise,
self-contained summary of:
- component ids and descriptions to expose
- files/directories to scan
- algorithms needed for discovery
- ast-grep commands or patterns needed
- generated provider contract details it must preserve
Do not edit files in this analysis step.`
}

function providerPrompt(spec, path) {
	return `Intent: ${spec.intent ?? spec.description}

Finish the generated ${spec.name} provider in ${path}.
Start by reading the AGENT INSTRUCTIONS and AGENT TODO comments in that file;
they are intentionally written as an implementation checklist for a small model.
Use the injected kit runtime object; do not import Kit internals from provider code.
Inspect unfamiliar APIs dynamically with kit.methods(), kit.method(name).signature(),
kit.method(name).source(), env.methods(), and env.method(name).parameterNames().
If you need to inspect the Kit API outside provider execution, use bun run kit repl.
The provider module must default-export function provider(kit) and return an object with:
- name(): provider name string
- async *types(): yields component type objects
- async *components(): yields existing component objects
- create(spec, env): delegates generation to a component type

Component type objects must implement:
- id(): component type id
- description(): user-facing type description
- schema(): TypeBox schema using kit.Type
- parse(argv): kit.parseArgs with kit.parseArgsOptionsFromSchema(this.schema())
- describe(spec): concise generated component description
- async *create(spec, env): calls env.createFile/editFile and yields kit.Event file/plan events

Component objects must implement:
- provider(): provider name
- id(): component id within provider
- description(): user-facing component description
- inspect(): structured fields including files when applicable

Available injected kit helpers include Type, Event, FileURI, Identifier, parseArgs,
parseArgsOptionsFromSchema, TableFormatter, and PlanFormatter.
Do not refactor unrelated code.`
}

function className(name) {
	return name
		.split('-')
		.map((part) => part[0].toUpperCase() + part.slice(1))
		.join('')
}

async function discoveryRoot(kit) {
	try {
		return await kit.repoRoot()
	} catch {
		return kit.FileURI.fromPath(process.cwd())
	}
}

async function exists(file) {
	try {
		return (await stat(file.path())).isDirectory()
	} catch {
		return false
	}
}

export default function provider(kit) {
	return new KitProviderProvider(kit)
}
